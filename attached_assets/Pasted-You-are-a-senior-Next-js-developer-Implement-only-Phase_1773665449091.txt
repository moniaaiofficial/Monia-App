### 1. DATABASE TABLES (Add to Supabase) – OPTIMIZED VERSION
```sql
-- Chats table (only if not exists)
CREATE TABLE IF NOT EXISTS chats (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  participants UUID[] NOT NULL,
  last_message TEXT,
  last_message_time TIMESTAMP DEFAULT NOW(),
  created_at TIMESTAMP DEFAULT NOW()
);

-- Messages table (only if not exists)
CREATE TABLE IF NOT EXISTS messages (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  chat_id UUID REFERENCES chats(id) ON DELETE CASCADE,
  sender_id UUID NOT NULL,
  content TEXT NOT NULL,
  type TEXT DEFAULT 'text',
  status TEXT CHECK (status IN ('sent','delivered','read')) DEFAULT 'sent',
  created_at TIMESTAMP DEFAULT NOW()
);

-- ⚡ CRITICAL for stable Realtime updates
ALTER TABLE messages REPLICA IDENTITY FULL;

-- Indexes for performance (safe to create even if exist)
CREATE INDEX IF NOT EXISTS idx_messages_chat_id ON messages(chat_id);
CREATE INDEX IF NOT EXISTS idx_messages_created_at ON messages(created_at);
CREATE INDEX IF NOT EXISTS idx_messages_sender_id ON messages(sender_id);
